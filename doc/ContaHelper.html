def datatable(dados = nil, columns = [])
  
  if columns.count > 0
    col_cab = []
    columns.each do |col|
      col_cab << tag(:th, {:value => col,:class => "text-center", "scope" => "col"})
    end
    line_cab = tag(:tr, :value => col_cab)
    tab_head = tag(:thead,{:value => line_cab, :class => "thead-dark"})
  end
  
  tag(:table, {:class => "table table-sm table-bordered table-hover", :value => tab_head}) 
end
